%--- Research Projects -------------------------------------------------------
\section{Research Projects}

\entrytitle{Memory-Efficient DNN Inference Compilation} \entrymeta{Mar 2020 -- Aug 2024}
\begin{itemize}
  \item \textbf{[Project Lead (6-person) / Collaborator]} Compile-time memory optimization that reduces peak GPU memory usage during single-model DNN inference.
  \begin{itemize}
    \item Designed an MLIR/GPU memory pool compiler with tensor liveness, layer fusion, and tensor coalescing.
    \item Co-developed skip-connection substitution and activation-layer fusion for tensor-decomposed models.
  \end{itemize}
  \item Result: \refcite{conf:temco,conf:occamy}; 1 patent filed; 34.6\% memory reduction with 1.25$\times$ speedup over PyTorch, 75.7\% internal-tensor peak-memory reduction on 10 models.
\end{itemize}

\vspace{0.4em}

\entrytitle{Scheduling for DNN Inference} \entrymeta{Mar 2023 -- Feb 2026}
\begin{itemize}
  \item \textbf{[Project Lead (7-person) / Project Lead (4-person)]} Explicit scheduling for DNN inference spanning task-level runtime scheduling and kernel-level compiler DSL.
  \begin{itemize}
    \item Designed Tailor-and-Stitch partitioning, shared tensor pool, and yield execution for multi-tenant inference.
    \item Designed a Triton-based intra/inter-kernel scheduling DSL for HW-compiler co-design.
  \end{itemize}
  \item Result: \refcite{jour:quilt,poster:quilt}; 2 patents filed; 25.4\%/37.7\% latency reduction and 10.8\% faster than Triton.
\end{itemize}

\vspace{0.4em}

\entrytitle{Approximation-based PIM Acceleration} \entrymeta{[Sub-project] 2024 -- 2025}
\begin{itemize}
  \item \textbf{[Collaborator]} Approximation-based image filtering eliminating inter-PE communication on commodity DRAM-PIM.
  \begin{itemize}
    \item Contributed to kernel scheduling for frequency-aware Input/Output Approximation and Boundary Duplication.
  \end{itemize}
  \item Result: \refcite{poster:pim}; 2 patents filed; 7.22$\times$ vs.\ PIM baseline and 5.23$\times$ vs.\ RTX 4090 on UPMEM, preserving 98.03\% accuracy.
\end{itemize}

\vspace{0.4em}

\entrytitle{Compiler for Privacy-Preserving Computing} \entrymeta{[Sub-project] 2020 -- 2024}
\begin{itemize}
  \item \textbf{[Collaborator]} Programming-language and compiler for encrypted computation on serverless IoT and edge-cloud.
  \begin{itemize}
    \item Co-developed a compiler coloring algorithm that selects HE vs.\ symmetric encryption by semantics.
    \item Contributed to a privacy-authority type system (Edge/Cloud/All/None) for type-directed FHE partitioning.
  \end{itemize}
  \item Result: \refcite{jour:pset,poster:pset,jour:selectivecrypt}; 1 patent granted; 1.59$\times$ faster response with 9.61$\times$ less communication, 4.92$\times$ speedup over partitioning-unaware FHE compilers.
\end{itemize}
